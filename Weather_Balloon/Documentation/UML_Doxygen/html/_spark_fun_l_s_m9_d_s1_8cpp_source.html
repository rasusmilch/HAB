183 void LSM9DS1::initGyro()
184 {
185  uint8_t tempRegValue = 0;
186 
187  // CTRL_REG1_G (Default value: 0x00)
188  // [ODR_G2][ODR_G1][ODR_G0][FS_G1][FS_G0][0][BW_G1][BW_G0]
189  // ODR_G[2:0] - Output data rate selection
190  // FS_G[1:0] - Gyroscope full-scale selection
191  // BW_G[1:0] - Gyroscope bandwidth selection
192 
193  // To disable gyro, set sample rate bits to 0. We'll only set sample
194  // rate if the gyro is enabled.
195  if (settings.gyro.enabled)
196  {
197  tempRegValue = (settings.gyro.sampleRate & 0x07) << 5;
198  }
199  switch (settings.gyro.scale)
200  {
201  case 500:
202  tempRegValue |= (0x1 << 3);
203  break;
204  case 2000:
205  tempRegValue |= (0x3 << 3);
206  break;
207  // Otherwise we'll set it to 245 dps (0x0 << 4)
208  }
209  tempRegValue |= (settings.gyro.bandwidth & 0x3);
210  xgWriteByte(CTRL_REG1_G, tempRegValue);
211 
212  // CTRL_REG2_G (Default value: 0x00)
213  // [0][0][0][0][INT_SEL1][INT_SEL0][OUT_SEL1][OUT_SEL0]
214  // INT_SEL[1:0] - INT selection configuration
215  // OUT_SEL[1:0] - Out selection configuration
216  xgWriteByte(CTRL_REG2_G, 0x00);
217 
218  // CTRL_REG3_G (Default value: 0x00)
219  // [LP_mode][HP_EN][0][0][HPCF3_G][HPCF2_G][HPCF1_G][HPCF0_G]
220  // LP_mode - Low-power mode enable (0: disabled, 1: enabled)
221  // HP_EN - HPF enable (0:disabled, 1: enabled)
222  // HPCF_G[3:0] - HPF cutoff frequency
223  tempRegValue = settings.gyro.lowPowerEnable ? (1<<7) : 0;
224  if (settings.gyro.HPFEnable)
225  {
226  tempRegValue |= (1<<6) | (settings.gyro.HPFCutoff & 0x0F);
227  }
228  xgWriteByte(CTRL_REG3_G, tempRegValue);
229 
230  // CTRL_REG4 (Default value: 0x38)
231  // [0][0][Zen_G][Yen_G][Xen_G][0][LIR_XL1][4D_XL1]
232  // Zen_G - Z-axis output enable (0:disable, 1:enable)
233  // Yen_G - Y-axis output enable (0:disable, 1:enable)
234  // Xen_G - X-axis output enable (0:disable, 1:enable)
235  // LIR_XL1 - Latched interrupt (0:not latched, 1:latched)
236  // 4D_XL1 - 4D option on interrupt (0:6D used, 1:4D used)
237  tempRegValue = 0;
238  if (settings.gyro.enableZ) tempRegValue |= (1<<5);
239  if (settings.gyro.enableY) tempRegValue |= (1<<4);
240  if (settings.gyro.enableX) tempRegValue |= (1<<3);
241  if (settings.gyro.latchInterrupt) tempRegValue |= (1<<1);
242  xgWriteByte(CTRL_REG4, tempRegValue);
243 
244  // ORIENT_CFG_G (Default value: 0x00)
245  // [0][0][SignX_G][SignY_G][SignZ_G][Orient_2][Orient_1][Orient_0]
246  // SignX_G - Pitch axis (X) angular rate sign (0: positive, 1: negative)
247  // Orient [2:0] - Directional user orientation selection
248  tempRegValue = 0;
249  if (settings.gyro.flipX) tempRegValue |= (1<<5);
250  if (settings.gyro.flipY) tempRegValue |= (1<<4);
251  if (settings.gyro.flipZ) tempRegValue |= (1<<3);
252  xgWriteByte(ORIENT_CFG_G, tempRegValue);
253 }

258 
259  // CTRL_REG5_XL (0x1F) (Default value: 0x38)
260  // [DEC_1][DEC_0][Zen_XL][Yen_XL][Zen_XL][0][0][0]
261  // DEC[0:1] - Decimation of accel data on OUT REG and FIFO.
262  // 00: None, 01: 2 samples, 10: 4 samples 11: 8 samples
263  // Zen_XL - Z-axis output enabled
264  // Yen_XL - Y-axis output enabled
265  // Xen_XL - X-axis output enabled
266  if (settings.accel.enableZ) tempRegValue |= (1<<5);
267  if (settings.accel.enableY) tempRegValue |= (1<<4);
268  if (settings.accel.enableX) tempRegValue |= (1<<3);
269 
270  xgWriteByte(CTRL_REG5_XL, tempRegValue);
271 
272  // CTRL_REG6_XL (0x20) (Default value: 0x00)
273  // [ODR_XL2][ODR_XL1][ODR_XL0][FS1_XL][FS0_XL][BW_SCAL_ODR][BW_XL1][BW_XL0]
274  // ODR_XL[2:0] - Output data rate & power mode selection
275  // FS_XL[1:0] - Full-scale selection
276  // BW_SCAL_ODR - Bandwidth selection
277  // BW_XL[1:0] - Anti-aliasing filter bandwidth selection
278  tempRegValue = 0;
279  // To disable the accel, set the sampleRate bits to 0.
280  if (settings.accel.enabled)
281  {
282  tempRegValue |= (settings.accel.sampleRate & 0x07) << 5;
283  }
284  switch (settings.accel.scale)
285  {
286  case 4:
287  tempRegValue |= (0x2 << 3);
288  break;
289  case 8:
290  tempRegValue |= (0x3 << 3);
291  break;
292  case 16:

294  break;

306  // HR - High resolution mode (0: disable, 1: enable)
307  // DCF[1:0] - Digital filter cutoff frequency
308  // FDS - Filtered data selection
309  // HPIS1 - HPF enabled for interrupt function
310  tempRegValue = 0;
311  if (settings.accel.highResEnable)
312  {
313  tempRegValue |= (1<<7); // Set HR bit

315  }

319 // This is a function that uses the FIFO to accumulate sample of accelerometer and gyro data, average
320 // them, scales them to gs and deg/s, respectively, and then passes the biases to the main sketch
321 // for subtraction from all subsequent data. There are no gyro and accelerometer bias registers to store
322 // the data as there are in the ADXL345, a precursor to the LSM9DS0, or the MPU-9150, so we have to
323 // subtract the biases ourselves. This results in a more accurate measurement in general and can
324 // remove errors due to imprecise or varying initial placement. Calibration of sensor data in this manner
325 // is good practice.
326 void LSM9DS1::calibrate(bool autoCalc)
327 {
328  uint8_t data[6] = {0, 0, 0, 0, 0, 0};
329  uint8_t samples = 0;
330  int ii;
331  int32_t aBiasRawTemp[3] = {0, 0, 0};
332  int32_t gBiasRawTemp[3] = {0, 0, 0};
333 
334  // Turn on FIFO and set threshold to 32 samples
335  enableFIFO(true);
336  setFIFO(FIFO_THS, 0x1F);
337  while (samples < 0x1F)
338  {
339  samples = (xgReadByte(FIFO_SRC) & 0x3F); // Read number of stored samples
340  }
341  for(ii = 0; ii < samples ; ii++)
342  { // Read the gyro data stored in the FIFO
343  readGyro();
344  gBiasRawTemp[0] += gx;
345  gBiasRawTemp[1] += gy;
346  gBiasRawTemp[2] += gz;
347  readAccel();
348  aBiasRawTemp[0] += ax;
349  aBiasRawTemp[1] += ay;
350  aBiasRawTemp[2] += az - (int16_t)(1./aRes); // Assumes sensor facing up!
351  }
352  for (ii = 0; ii < 3; ii++)
353  {
354  gBiasRaw[ii] = gBiasRawTemp[ii] / samples;
355  gBias[ii] = calcGyro(gBiasRaw[ii]);
356  aBiasRaw[ii] = aBiasRawTemp[ii] / samples;
357  aBias[ii] = calcAccel(aBiasRaw[ii]);
358  }
359 
360  enableFIFO(false);
361  setFIFO(FIFO_OFF, 0x00);
362 
363  if (autoCalc) _autoCalc = true;
364 }

421  tempRegValue |= (settings.mag.sampleRate & 0x7) << 2;

424  // CTRL_REG2_M (Default value 0x00)

427  // REBOOT - Reboot memory content (0:normal, 1:reboot)

432  case 8:
433  tempRegValue |= (0x1 << 5);
434  break;
435  case 12:
436  tempRegValue |= (0x2 << 5);
437  break;

439  tempRegValue |= (0x3 << 5);
