539  // CRC16 code via Scott Dattalo www.dattalo.com

542  x ^= x >> 4;
543  crc = (crc << 8) ^ (x << 12) ^ (x << 5) ^ x;
544  }

583  //Pin change interrupt for Geiger counters.
584  //No falling or rising edge detection on these pins so we
585  // need to track previous pin states.
586  // When a pin has changed value we update the appropriate counter.
587  // Geiger counter outputs a low pulse of 150 microseconds on particle detection.
588 
589 
590  // Previous pin value for port. Static to preserve between calls.
591  static uint8_t previous_pins_j = 0;
592 
593  // Grab current pin state quickly. Low pulse from Geiger is only 150 microseconds.
594  uint8_t current_state_j = PINJ;
595 
596  // If the current pin state is low, and previous it was high, we have
597  // a falling edge.
598  if ((current_state_j & (1<<PJ1)) == 0 && (previous_pins_j & (1<<PJ1)) > 0) {
599  // Trigger count
600  geiger.y_count++;
601  }
602 
603  if ((current_state_j & (1<<PJ0)) == 0 && (previous_pins_j & (1<<PJ0)) > 0) {
604  // Trigger count
605  geiger.x_count++;
606  }
607 
608  // Store for later.
609  previous_pins_j = current_state_j;
