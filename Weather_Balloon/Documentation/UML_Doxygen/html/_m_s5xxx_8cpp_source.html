66 unsigned int MS5xxx::Calc_CRC4(unsigned char poly)
67 {
68  int cnt; // simple counter
69  unsigned int n_rem; // CRC remainder
70  unsigned int crc_read; // original value of the CRC
71  unsigned int l_pol = poly;
72  unsigned char n_bit;
73 
74  l_pol = ( l_pol << 8 ) & 0xf000; // shift bits and apply mask
75  n_rem = 0x0000;
76 
77  crc_read = C[ 7 ]; // save read RCR
78  C[ 7 ] = ( 0xFF00 & ( C[ 7 ] ) ); // CRC byte is replaced by 0
79  for ( cnt = 0; cnt < 16; cnt++ ) // operation is performed on bytes
80  {// choose LSB or MSB
81  if ( cnt % 2 == 1 ) n_rem ^= ( unsigned short ) ( ( C[ cnt >> 1 ] ) & 0x00FF );
82  else n_rem ^= ( unsigned short ) ( ( C[ cnt >> 1 ] >> 8) & 0x00FF );
83 
84  for ( n_bit = 8; n_bit > 0; n_bit-- )
85  {
86  if ( n_rem & ( 0x8000 ) )
87  {
88  n_rem = ( n_rem << 1 ) ^ l_pol;
89  }
90  else
91  {
92  n_rem = ( n_rem << 1 );
93  }
94  }
95  }
96  C[ 7 ] = crc_read;
97  n_rem = (0x000F & (n_rem >> 12)); // final 4-bit remainder is CRC code
98  return n_rem;
99 }

116 unsigned long MS5xxx::read_adc(unsigned char aCMD)
117 {
118  unsigned long value=0;
119  unsigned long c=0;
120 
121  send_cmd(MS5xxx_CMD_ADC_CONV+aCMD); // start DAQ and conversion of ADC data
122  switch (aCMD & 0x0f)
123  {
124  case MS5xxx_CMD_ADC_256 : delayMicroseconds(900);
125  break;
126  case MS5xxx_CMD_ADC_512 : delay(3);
127  break;
128  case MS5xxx_CMD_ADC_1024: delay(4);
129  break;
130  case MS5xxx_CMD_ADC_2048: delay(6);
131  break;
132  case MS5xxx_CMD_ADC_4096: delay(10);
133  break;
134  }
135  send_cmd(MS5xxx_CMD_ADC_READ); // read out values
136  _Wire->requestFrom(i2caddr, 3);
137  c = _Wire->read();
138  value = (c<<16);
139  c = _Wire->read();
140  value += (c<<8);
141  c = _Wire->read();
142  value += c;
143  _Wire->endTransmission(true);
144 
145  return value;
146 }

148 void MS5xxx::Readout() {
149  unsigned long D1=0, D2=0;
150 
151  double dT;
152  double OFF;
153  double SENS;
154 
155  D2=read_adc(MS5xxx_CMD_ADC_D2+MS5xxx_CMD_ADC_4096);
156  D1=read_adc(MS5xxx_CMD_ADC_D1+MS5xxx_CMD_ADC_4096);
157 
158  // calculate 1st order pressure and temperature (MS5607 1st order algorithm)
159  dT=D2-C[5]*pow(2,8);
160  OFF=C[2]*pow(2,17)+dT*C[4]/pow(2,6);
161  SENS=C[1]*pow(2,16)+dT*C[3]/pow(2,7);
162  TEMP=(2000+(dT*C[6])/pow(2,23));
163  P=(((D1*SENS)/pow(2,21)-OFF)/pow(2,15));
164 
165  // perform higher order corrections
166  double T2=0., OFF2=0., SENS2=0.;
167  if(TEMP<2000) {
168  T2=dT*dT/pow(2,31);
169  OFF2=61*(TEMP-2000)*(TEMP-2000)/pow(2,4);
170  SENS2=2*(TEMP-2000)*(TEMP-2000);
171  if(TEMP<-1500) {
172  OFF2+=15*(TEMP+1500)*(TEMP+1500);
173  SENS2+=8*(TEMP+1500)*(TEMP+1500);
174  }
175  }
176 
177  TEMP-=T2;
178  OFF-=OFF2;
179  SENS-=SENS2;
180  P=(((D1*SENS)/pow(2,21)-OFF)/pow(2,15));
181 }
