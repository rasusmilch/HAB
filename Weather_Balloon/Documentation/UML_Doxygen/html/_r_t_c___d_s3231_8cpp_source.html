101 float RTC_DS3231::getTempAsFloat()
102 {
103  //set the address pointer in preparation for read
104  Wire.beginTransmission(DS3231_ADDRESS);
105  Wire.SEND( DS3231_REG_TEMP_MSB ); // was just '0'
106  Wire.endTransmission();
107 
108  Wire.requestFrom(DS3231_ADDRESS, 2);
109  int8_t sig = Wire.RECEIVE(); //signed MSB
110  uint8_t fract = Wire.RECEIVE() & 0b11000000; //rest should be zeroes anyway.
111 
112  fract = fract >> 6; // now in 2 lsb's
113  float temp = (float)fract * 0.25; // total up, .00, .25, .50
114  if(sig < 0) {
115  temp = sig - temp; // calculate the fract correctly
116  } else {
117  temp = sig + temp; // add 25 + .25 = 25.25
118  }
119  return temp;
120 }

198 void RTC_DS3231::forceTempConv(uint8_t block)
199 {
200  Wire.beginTransmission(DS3231_ADDRESS);
201  Wire.SEND( DS3231_REG_CONTROL );
202  Wire.endTransmission();
203 
204  // control register
205  Wire.requestFrom(DS3231_ADDRESS, 1);
206 
207  uint8_t creg = Wire.RECEIVE(); // do we need the bcd2bin
208 
209  creg |= 0b00100000; // Write CONV bit
210 
211  Wire.beginTransmission(DS3231_ADDRESS);
212  Wire.SEND(DS3231_REG_CONTROL);
213  Wire.SEND(creg);
214  Wire.endTransmission();
215 
216  do
217  {
218  // Block until CONV is 0
219  Wire.beginTransmission(DS3231_ADDRESS);
220  Wire.SEND(DS3231_REG_CONTROL);
221  Wire.endTransmission();
222  Wire.requestFrom(DS3231_ADDRESS, 1);
223  } while ((block && (Wire.RECEIVE() & 0b00100000) != 0));
224 }
