65 static uint8_t spixfer(uint8_t clock, uint8_t miso, uint8_t mosi, uint8_t data) {
66  uint8_t reply = 0;
67  for (int i=7; i>=0; i--) {
68  reply <<= 1;
69  digitalWrite(clock, LOW);
70  digitalWrite(mosi, data & (1<<i));
71  digitalWrite(clock, HIGH);
72  if (digitalRead(miso))
73  reply |= 1;
74  }
75  return reply;
76 }

242 void Adafruit_ADXL345_Unified::setRange(range_t range)
243 {
244  /* Red the data format register to preserve bits */
245  uint8_t format = readRegister(ADXL345_REG_DATA_FORMAT);
246 
247  /* Update the data rate */
248  format &= ~0x0F;
249  format |= range;
250 
251  /* Make sure that the FULL-RES bit is enabled for range scaling */
252  format |= 0x08;
253 
254  /* Write the register back to the IC */
255  writeRegister(ADXL345_REG_DATA_FORMAT, format);
256 
257  /* Keep track of the current range (to avoid readbacks) */
258  _range = range;
259 }
