178 static bool hdlcParse(Hdlc *hdlc, bool bit, FIFOBuffer *fifo) {
179  // Initialise a return value. We start with the
180  // assumption that all is going to end well :)
181  bool ret = true;
182 
183  // Bitshift our byte of demodulated bits to
184  // the left by one bit, to make room for the
185  // next incoming bit
186  hdlc->demodulatedBits <<= 1;
187  // And then put the newest bit from the
188  // demodulator into the byte.
189  hdlc->demodulatedBits |= bit ? 1 : 0;
190 
191  // Now we'll look at the last 8 received bits, and
192  // check if we have received a HDLC flag (01111110)
193  if (hdlc->demodulatedBits == HDLC_FLAG) {
194  // If we have, check that our output buffer is
195  // not full.
196  if (!fifo_isfull(fifo)) {
197  // If it isn't, we'll push the HDLC_FLAG into
198  // the buffer and indicate that we are now
199  // receiving data. For bling we also turn
200  // on the RX LED.
201  fifo_push(fifo, HDLC_FLAG);
202  hdlc->receiving = true;
203  if(!LibAPRS_open_squelch) {
204  LED_RX_ON();
205  }
206  } else {
207  // If the buffer is full, we have a problem
208  // and abort by setting the return value to
209  // false and stopping the here.
210 
211  ret = false;
212  hdlc->receiving = false;
213  LED_RX_OFF();
214  }
215 
216  // Everytime we receive a HDLC_FLAG, we reset the
217  // storage for our current incoming byte and bit
218  // position in that byte. This effectively
219  // synchronises our parsing to the start and end
220  // of the received bytes.
221  hdlc->currentByte = 0;
222  hdlc->bitIndex = 0;
223  return ret;
224  }
225 
226  // Check if we have received a RESET flag (01111111)
227  // In this comparison we also detect when no transmission
228  // (or silence) is taking place, and the demodulator
229  // returns an endless stream of zeroes. Due to the NRZ
230  // coding, the actual bits send to this function will
231  // be an endless stream of ones, which this AND operation
232  // will also detect.
233  if ((hdlc->demodulatedBits & HDLC_RESET) == HDLC_RESET) {
234  // If we have, something probably went wrong at the
235  // transmitting end, and we abort the reception.
236  hdlc->receiving = false;
237  LED_RX_OFF();
238  return ret;
239  }
240 
241  // If we have not yet seen a HDLC_FLAG indicating that
242  // a transmission is actually taking place, don't bother
243  // with anything.
244  if (!hdlc->receiving)
245  return ret;
246 
247  // First check if what we are seeing is a stuffed bit.
248  // Since the different HDLC control characters like
249  // HDLC_FLAG, HDLC_RESET and such could also occur in
250  // a normal data stream, we employ a method known as
251  // "bit stuffing". All control characters have more than
252  // 5 ones in a row, so if the transmitting party detects
253  // this sequence in the _data_ to be transmitted, it inserts
254  // a zero to avoid the receiving party interpreting it as
255  // a control character. Therefore, if we detect such a
256  // "stuffed bit", we simply ignore it and wait for the
257  // next bit to come in.
258  //
259  // We do the detection by applying an AND bit-mask to the
260  // stream of demodulated bits. This mask is 00111111 (0x3f)
261  // if the result of the operation is 00111110 (0x3e), we
262  // have detected a stuffed bit.
263  if ((hdlc->demodulatedBits & 0x3f) == 0x3e)
264  return ret;
265 
266  // If we have an actual 1 bit, push this to the current byte
267  // If it's a zero, we don't need to do anything, since the
268  // bit is initialized to zero when we bitshifted earlier.
269  if (hdlc->demodulatedBits & 0x01)
270  hdlc->currentByte |= 0x80;
271 
272  // Increment the bitIndex and check if we have a complete byte
273  if (++hdlc->bitIndex >= 8) {
274  // If we have a HDLC control character, put a AX.25 escape
275  // in the received data. We know we need to do this,
276  // because at this point we must have already seen a HDLC
277  // flag, meaning that this control character is the result
278  // of a bitstuffed byte that is equal to said control
279  // character, but is actually part of the data stream.
280  // By inserting the escape character, we tell the protocol
281  // layer that this is not an actual control character, but
282  // data.
283  if ((hdlc->currentByte == HDLC_FLAG ||
284  hdlc->currentByte == HDLC_RESET ||
285  hdlc->currentByte == AX25_ESC)) {
286  // We also need to check that our received data buffer
287  // is not full before putting more data in
288  if (!fifo_isfull(fifo)) {
289  fifo_push(fifo, AX25_ESC);
290  } else {
291  // If it is, abort and return false
292  hdlc->receiving = false;
293  LED_RX_OFF();
294  ret = false;
295  }
296  }
297 
298  // Push the actual byte to the received data FIFO,
299  // if it isn't full.
300  if (!fifo_isfull(fifo)) {
301  fifo_push(fifo, hdlc->currentByte);
302  } else {
303  // If it is, well, you know by now!
304  hdlc->receiving = false;
305  LED_RX_OFF();
306  ret = false;
307  }
308 
309  // Wipe received byte and reset bit index to 0
310  hdlc->currentByte = 0;
311  hdlc->bitIndex = 0;
312 
313  } else {
314  // We don't have a full byte yet, bitshift the byte
315  // to make room for the next bit
316  hdlc->currentByte >>= 1;
317  }
318 
319  //digitalWrite(13, LOW);
320  return ret;
321 }

324 void AFSK_adc_isr(Afsk *afsk, int8_t currentSample) {
325  // To determine the received frequency, and thereby
326  // the bit of the sample, we multiply the sample by
327  // a sample delayed by (samples per bit / 2).
328  // We then lowpass-filter the samples with a
329  // Chebyshev filter. The lowpass filtering serves
330  // to "smooth out" the variations in the samples.
331 
332  afsk->iirX[0] = afsk->iirX[1];
333  afsk->iirX[1] = ((int8_t)fifo_pop(&afsk->delayFifo) * currentSample) >> 2;
334 
335  afsk->iirY[0] = afsk->iirY[1];
336 
337  afsk->iirY[1] = afsk->iirX[0] + afsk->iirX[1] + (afsk->iirY[0] >> 1); // Chebyshev filter
338 
339 
340  // We put the sampled bit in a delay-line:
341  // First we bitshift everything 1 left
342  afsk->sampledBits <<= 1;
343  // And then add the sampled bit to our delay line
344  afsk->sampledBits |= (afsk->iirY[1] > 0) ? 1 : 0;
345 
346  // Put the current raw sample in the delay FIFO
347  fifo_push(&afsk->delayFifo, currentSample);
348 
349  // We need to check whether there is a signal transition.
350  // If there is, we can recalibrate the phase of our
351  // sampler to stay in sync with the transmitter. A bit of
352  // explanation is required to understand how this works.
353  // Since we have PHASE_MAX/PHASE_BITS = 8 samples per bit,
354  // we employ a phase counter (currentPhase), that increments
355  // by PHASE_BITS everytime a sample is captured. When this
356  // counter reaches PHASE_MAX, it wraps around by modulus
357  // PHASE_MAX. We then look at the last three samples we
358  // captured and determine if the bit was a one or a zero.
359  //
360  // This gives us a "window" looking into the stream of
361  // samples coming from the ADC. Sort of like this:
362  //
363  // Past Future
364  // 0000000011111111000000001111111100000000
365  // |________|
366  // ||
367  // Window
368  //
369  // Every time we detect a signal transition, we adjust
370  // where this window is positioned little. How much we
371  // adjust it is defined by PHASE_INC. If our current phase
372  // phase counter value is less than half of PHASE_MAX (ie,
373  // the window size) when a signal transition is detected,
374  // add PHASE_INC to our phase counter, effectively moving
375  // the window a little bit backward (to the left in the
376  // illustration), inversely, if the phase counter is greater
377  // than half of PHASE_MAX, we move it forward a little.
378  // This way, our "window" is constantly seeking to position
379  // it's center at the bit transitions. Thus, we synchronise
380  // our timing to the transmitter, even if it's timing is
381  // a little off compared to our own.
382  if (SIGNAL_TRANSITIONED(afsk->sampledBits)) {
383  if (afsk->currentPhase < PHASE_THRESHOLD) {
384  afsk->currentPhase += PHASE_INC;
385  } else {
386  afsk->currentPhase -= PHASE_INC;
387  }
388  }
389 
390  // We increment our phase counter
391  afsk->currentPhase += PHASE_BITS;
392 
393  // Check if we have reached the end of
394  // our sampling window.
395  if (afsk->currentPhase >= PHASE_MAX) {
396  // If we have, wrap around our phase
397  // counter by modulus
398  afsk->currentPhase %= PHASE_MAX;
399 
400  // Bitshift to make room for the next
401  // bit in our stream of demodulated bits
402  afsk->actualBits <<= 1;
403 
404  // We determine the actual bit value by reading
405  // the last 3 sampled bits. If there is three or
406  // more 1's, we will assume that the transmitter
407  // sent us a one, otherwise we assume a zero
408  uint8_t bits = afsk->sampledBits & 0x07;
409  if (bits == 0x07 || // 111
410  bits == 0x06 || // 110
411  bits == 0x05 || // 101
412  bits == 0x03 // 011
413  ) {
414  afsk->actualBits |= 1;
415  }
416 
418  // uint8_t bits = afsk->sampledBits & 0x0f;
419  // uint8_t c = 0;
420  // c += bits & BV(1);
421  // c += bits & BV(2);
422  // c += bits & BV(3);
423  // c += bits & BV(4);
424  // c += bits & BV(5);
425  // if (c >= 3) afsk->actualBits |= 1;
427 
428  // Now we can pass the actual bit to the HDLC parser.
429  // We are using NRZ coding, so if 2 consecutive bits
430  // have the same value, we have a 1, otherwise a 0.
431  // We use the TRANSITION_FOUND function to determine this.
432  //
433  // This is smart in combination with bit stuffing,
434  // since it ensures a transmitter will never send more
435  // than five consecutive 1's. When sending consecutive
436  // ones, the signal stays at the same level, and if
437  // this happens for longer periods of time, we would
438  // not be able to synchronize our phase to the transmitter
439  // and would start experiencing "bit slip".
440  //
441  // By combining bit-stuffing with NRZ coding, we ensure
442  // that the signal will regularly make transitions
443  // that we can use to synchronize our phase.
444  //
445  // We also check the return of the Link Control parser
446  // to check if an error occured.
447 
448  if (!hdlcParse(&afsk->hdlc, !TRANSITION_FOUND(afsk->actualBits), &afsk->rxFifo)) {
449  afsk->status |= 1;
450  if (fifo_isfull(&afsk->rxFifo)) {
451  fifo_flush(&afsk->rxFifo);
452  afsk->status = 0;
453  }
454  }
455  }
