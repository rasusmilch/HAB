138 void Adafruit_TSL2561_Unified::getData (uint16_t *broadband, uint16_t *ir)
139 {
140  /* Enable the device by setting the control bit to 0x03 */
141  enable();
142 
143  /* Wait x ms for ADC to complete */
144  switch (_tsl2561IntegrationTime)
145  {
146  case TSL2561_INTEGRATIONTIME_13MS:
147  delay(TSL2561_DELAY_INTTIME_13MS); // KTOWN: Was 14ms
148  break;
149  case TSL2561_INTEGRATIONTIME_101MS:
150  delay(TSL2561_DELAY_INTTIME_101MS); // KTOWN: Was 102ms
151  break;
152  default:
153  delay(TSL2561_DELAY_INTTIME_402MS); // KTOWN: Was 403ms
154  break;
155  }
156 
157  /* Reads a two byte value from channel 0 (visible + infrared) */
158  *broadband = read16(TSL2561_COMMAND_BIT | TSL2561_WORD_BIT | TSL2561_REGISTER_CHAN0_LOW);
159 
160  /* Reads a two byte value from channel 1 (infrared) */
161  *ir = read16(TSL2561_COMMAND_BIT | TSL2561_WORD_BIT | TSL2561_REGISTER_CHAN1_LOW);
162 
163  /* Turn the device off to save power */
164  disable();

279 void Adafruit_TSL2561_Unified::getLuminosity (uint16_t *broadband, uint16_t *ir)
280 {
281  bool valid = false;
282 
283  if (!_tsl2561Initialised) begin();
284 
285  /* If Auto gain disabled get a single reading and continue */
286  if(!_tsl2561AutoGain)
287  {
288  getData (broadband, ir);
289  return;
290  }
291 
292  /* Read data until we find a valid range */
293  bool _agcCheck = false;
294  do
295  {
296  uint16_t _b, _ir;
297  uint16_t _hi, _lo;
298  tsl2561IntegrationTime_t _it = _tsl2561IntegrationTime;
299 
300  /* Get the hi/low threshold for the current integration time */
301  switch(_it)
302  {
303  case TSL2561_INTEGRATIONTIME_13MS:
304  _hi = TSL2561_AGC_THI_13MS;
305  _lo = TSL2561_AGC_TLO_13MS;
306  break;
307  case TSL2561_INTEGRATIONTIME_101MS:
308  _hi = TSL2561_AGC_THI_101MS;
309  _lo = TSL2561_AGC_TLO_101MS;
310  break;
311  default:
312  _hi = TSL2561_AGC_THI_402MS;
313  _lo = TSL2561_AGC_TLO_402MS;
314  break;
315  }
316 
317  getData(&_b, &_ir);
318 
319  /* Run an auto-gain check if we haven't already done so ... */
320  if (!_agcCheck)
321  {
322  if ((_b < _lo) && (_tsl2561Gain == TSL2561_GAIN_1X))
323  {
324  /* Increase the gain and try again */
325  setGain(TSL2561_GAIN_16X);
326  /* Drop the previous conversion results */
327  getData(&_b, &_ir);
328  /* Set a flag to indicate we've adjusted the gain */
329  _agcCheck = true;
330  }
331  else if ((_b > _hi) && (_tsl2561Gain == TSL2561_GAIN_16X))
332  {
333  /* Drop gain to 1x and try again */
334  setGain(TSL2561_GAIN_1X);
335  /* Drop the previous conversion results */
336  getData(&_b, &_ir);
337  /* Set a flag to indicate we've adjusted the gain */
338  _agcCheck = true;
339  }
340  else
341  {
342  /* Nothing to look at here, keep moving ....
343  Reading is either valid, or we're already at the chips limits */
344  *broadband = _b;
345  *ir = _ir;
346  valid = true;
347  }
348  }
349  else
350  {
351  /* If we've already adjusted the gain once, just return the new results.
352  This avoids endless loops where a value is at one extreme pre-gain,
353  and the the other extreme post-gain */
354  *broadband = _b;
355  *ir = _ir;
356  valid = true;
357  }
358  } while (!valid);
359 }

367 uint32_t Adafruit_TSL2561_Unified::calculateLux(uint16_t broadband, uint16_t ir)
368 {
369  unsigned long chScale;
370  unsigned long channel1;
371  unsigned long channel0;
372 
373  /* Make sure the sensor isn't saturated! */
374  uint16_t clipThreshold;
375  switch (_tsl2561IntegrationTime)
376  {
377  case TSL2561_INTEGRATIONTIME_13MS:
378  clipThreshold = TSL2561_CLIPPING_13MS;
379  break;
380  case TSL2561_INTEGRATIONTIME_101MS:
381  clipThreshold = TSL2561_CLIPPING_101MS;
382  break;
383  default:
384  clipThreshold = TSL2561_CLIPPING_402MS;
385  break;
386  }
387 
388  /* Return 65536 lux if the sensor is saturated */
389  if ((broadband > clipThreshold) || (ir > clipThreshold))
390  {
391  return 65536;
392  }
393 
394  /* Get the correct scale depending on the intergration time */
395  switch (_tsl2561IntegrationTime)
396  {
397  case TSL2561_INTEGRATIONTIME_13MS:
398  chScale = TSL2561_LUX_CHSCALE_TINT0;
399  break;
400  case TSL2561_INTEGRATIONTIME_101MS:
401  chScale = TSL2561_LUX_CHSCALE_TINT1;
402  break;
403  default: /* No scaling ... integration time = 402ms */
404  chScale = (1 << TSL2561_LUX_CHSCALE);
405  break;
406  }
407 
408  /* Scale for gain (1x or 16x) */
409  if (!_tsl2561Gain) chScale = chScale << 4;
410 
411  /* Scale the channel values */
412  channel0 = (broadband * chScale) >> TSL2561_LUX_CHSCALE;
413  channel1 = (ir * chScale) >> TSL2561_LUX_CHSCALE;
414 
415  /* Find the ratio of the channel values (Channel1/Channel0) */
416  unsigned long ratio1 = 0;
417  if (channel0 != 0) ratio1 = (channel1 << (TSL2561_LUX_RATIOSCALE+1)) / channel0;
418 
419  /* round the ratio value */
420  unsigned long ratio = (ratio1 + 1) >> 1;
421 
422  unsigned int b, m;
423 
424 #ifdef TSL2561_PACKAGE_CS
425  if ((ratio >= 0) && (ratio <= TSL2561_LUX_K1C))
426  {b=TSL2561_LUX_B1C; m=TSL2561_LUX_M1C;}
427  else if (ratio <= TSL2561_LUX_K2C)
428  {b=TSL2561_LUX_B2C; m=TSL2561_LUX_M2C;}
429  else if (ratio <= TSL2561_LUX_K3C)
430  {b=TSL2561_LUX_B3C; m=TSL2561_LUX_M3C;}
431  else if (ratio <= TSL2561_LUX_K4C)
432  {b=TSL2561_LUX_B4C; m=TSL2561_LUX_M4C;}
433  else if (ratio <= TSL2561_LUX_K5C)
434  {b=TSL2561_LUX_B5C; m=TSL2561_LUX_M5C;}
435  else if (ratio <= TSL2561_LUX_K6C)
436  {b=TSL2561_LUX_B6C; m=TSL2561_LUX_M6C;}
437  else if (ratio <= TSL2561_LUX_K7C)
438  {b=TSL2561_LUX_B7C; m=TSL2561_LUX_M7C;}
439  else if (ratio > TSL2561_LUX_K8C)
440  {b=TSL2561_LUX_B8C; m=TSL2561_LUX_M8C;}
441 #else
442  if ((ratio >= 0) && (ratio <= TSL2561_LUX_K1T))
443  {b=TSL2561_LUX_B1T; m=TSL2561_LUX_M1T;}
444  else if (ratio <= TSL2561_LUX_K2T)
445  {b=TSL2561_LUX_B2T; m=TSL2561_LUX_M2T;}
446  else if (ratio <= TSL2561_LUX_K3T)
447  {b=TSL2561_LUX_B3T; m=TSL2561_LUX_M3T;}
448  else if (ratio <= TSL2561_LUX_K4T)
449  {b=TSL2561_LUX_B4T; m=TSL2561_LUX_M4T;}
450  else if (ratio <= TSL2561_LUX_K5T)
451  {b=TSL2561_LUX_B5T; m=TSL2561_LUX_M5T;}
452  else if (ratio <= TSL2561_LUX_K6T)
453  {b=TSL2561_LUX_B6T; m=TSL2561_LUX_M6T;}
454  else if (ratio <= TSL2561_LUX_K7T)
455  {b=TSL2561_LUX_B7T; m=TSL2561_LUX_M7T;}
456  else if (ratio > TSL2561_LUX_K8T)
457  {b=TSL2561_LUX_B8T; m=TSL2561_LUX_M8T;}
458 #endif
459 
460  unsigned long temp;
461  temp = ((channel0 * b) - (channel1 * m));
462 
463  /* Do not allow negative lux value */
464  if (temp < 0) temp = 0;
465 
466  /* Round lsb (2^(LUX_SCALE-1)) */
467  temp += (1 << (TSL2561_LUX_LUXSCALE-1));
468 
469  /* Strip off fractional portion */
470  uint32_t lux = temp >> TSL2561_LUX_LUXSCALE;
471 
472  /* Signal I2C had no errors */
473  return lux;
474 }
