115 bool DallasTemperature::readScratchPad(const uint8_t* deviceAddress, uint8_t* scratchPad){
116 
117  // send the reset command and fail fast
118  int b = _wire->reset();
119  if (b == 0) return false;
120 
121  _wire->select(deviceAddress);
122  _wire->write(READSCRATCH);
123 
124  // Read all registers in a simple loop
125  // byte 0: temperature LSB
126  // byte 1: temperature MSB
127  // byte 2: high alarm temp
128  // byte 3: low alarm temp
129  // byte 4: DS18S20: store for crc
130  // DS18B20 & DS1822: configuration register
131  // byte 5: internal use & crc
132  // byte 6: DS18S20: COUNT_REMAIN
133  // DS18B20 & DS1822: store for crc
134  // byte 7: DS18S20: COUNT_PER_C
135  // DS18B20 & DS1822: store for crc
136  // byte 8: SCRATCHPAD_CRC
137  for(uint8_t i = 0; i < 9; i++){
138  scratchPad[i] = _wire->read();
139  }
140 
141  b = _wire->reset();
142  return (b == 1);
143 }

146 void DallasTemperature::writeScratchPad(const uint8_t* deviceAddress, const uint8_t* scratchPad){
147 
148  _wire->reset();
149  _wire->select(deviceAddress);
150  _wire->write(WRITESCRATCH);
151  _wire->write(scratchPad[HIGH_ALARM_TEMP]); // high alarm temp
152  _wire->write(scratchPad[LOW_ALARM_TEMP]); // low alarm temp

156 
157  _wire->reset();
158

161  _wire->write(COPYSCRATCH, parasite);
162  delay(20); // <--- added 20ms delay to allow 10ms long EEPROM write operation (as specified by datasheet)
163 
164  if (parasite) delay(10); // 10ms delay
165  _wire->reset();

196 // set resolution of a device to 9, 10, 11, or 12 bits
197 // if new resolution is out of range, 9 bits is used.
198 bool DallasTemperature::setResolution(const uint8_t* deviceAddress, uint8_t newResolution, bool skipGlobalBitResolutionCalculation){
199 
200  // ensure same behavior as setResolution(uint8_t newResolution)
201  newResolution = constrain(newResolution, 9, 12);
202 
203  // return when stored value == new value
204  if(getResolution(deviceAddress) == newResolution) return true;
205 
206  ScratchPad scratchPad;
207  if (isConnected(deviceAddress, scratchPad)){
208 
209  // DS1820 and DS18S20 have no resolution configuration register
210  if (deviceAddress[0] != DS18S20MODEL){
211 
212  switch (newResolution){
213  case 12:
214  scratchPad[CONFIGURATION] = TEMP_12_BIT;
215  break;
216  case 11:
217  scratchPad[CONFIGURATION] = TEMP_11_BIT;
218  break;
219  case 10:
220  scratchPad[CONFIGURATION] = TEMP_10_BIT;
221  break;
222  case 9:
223  default:
224  scratchPad[CONFIGURATION] = TEMP_9_BIT;
225  break;
226  }
227  writeScratchPad(deviceAddress, scratchPad);
228 
229  // without calculation we can always set it to max
230  bitResolution = max(bitResolution, newResolution);
231 
232  if(!skipGlobalBitResolutionCalculation && (bitResolution > newResolution)){
233  bitResolution = newResolution;
234  DeviceAddress deviceAddr;
235  for (int i=0; i<devices; i++)
236  {
237  getAddress(deviceAddr, i);
238  bitResolution = max(bitResolution, getResolution(deviceAddr));
239  }
240  }
241  }
242  return true; // new value set
243  }
244 
245  return false;
246 
247 }

432 // reads scratchpad and returns fixed-point temperature, scaling factor 2^-7
433 int16_t DallasTemperature::calculateTemperature(const uint8_t* deviceAddress, uint8_t* scratchPad){
434 
435  int16_t fpTemperature =
436  (((int16_t) scratchPad[TEMP_MSB]) << 11) |
437  (((int16_t) scratchPad[TEMP_LSB]) << 3);
438 
439  /*
440  DS1820 and DS18S20 have a 9-bit temperature register.
441 
442  Resolutions greater than 9-bit can be calculated using the data from
443  the temperature, and COUNT REMAIN and COUNT PER C registers in the
444  scratchpad. The resolution of the calculation depends on the model.
445 
446  While the COUNT PER C register is hard-wired to 16 (10h) in a
447  DS18S20, it changes with temperature in DS1820.
448 
449  After reading the scratchpad, the TEMP_READ value is obtained by
450  truncating the 0.5C bit (bit 0) from the temperature data. The
451  extended resolution temperature can then be calculated using the
452  following equation:
453 
454  COUNT_PER_C - COUNT_REMAIN
455  TEMPERATURE = TEMP_READ - 0.25 + --------------------------
456  COUNT_PER_C
457 
458  Hagai Shatz simplified this to integer arithmetic for a 12 bits
459  value for a DS18S20, and James Cameron added legacy DS1820 support.
460 
461  See - http://myarduinotoy.blogspot.co.uk/2013/02/12bit-result-from-ds18s20.html
462  */
463 
464  if (deviceAddress[0] == DS18S20MODEL){
465  fpTemperature = ((fpTemperature & 0xfff0) << 3) - 16 +
466  (
467  ((scratchPad[COUNT_PER_C] - scratchPad[COUNT_REMAIN]) << 7) /
468  scratchPad[COUNT_PER_C]
469  );
470  }
471 
472  return fpTemperature;
473 }

684 // This is a modified version of the OneWire::search method.
685 //
686 // Also added the OneWire search fix documented here:
687 // http://www.arduino.cc/cgi-bin/yabb2/YaBB.pl?num=1238032295
688 //
689 // Perform an alarm search. If this function returns a '1' then it has
690 // enumerated the next device and you may retrieve the ROM from the
691 // OneWire::address variable. If there are no devices, no further
692 // devices, or something horrible happens in the middle of the
693 // enumeration then a 0 is returned. If a new device is found then
694 // its address is copied to newAddr. Use
695 // DallasTemperature::resetAlarmSearch() to start over.
696 bool DallasTemperature::alarmSearch(uint8_t* newAddr){
697 
698  uint8_t i;
699  char lastJunction = -1;
700  uint8_t done = 1;
701 
702  if (alarmSearchExhausted) return false;
703  if (!_wire->reset()) return false;
704 
705  // send the alarm search command
706  _wire->write(0xEC, 0);
707 
708  for(i = 0; i < 64; i++){
709 
710  uint8_t a = _wire->read_bit( );
711  uint8_t nota = _wire->read_bit( );
712  uint8_t ibyte = i / 8;
713  uint8_t ibit = 1 << (i & 7);
714 
715  // I don't think this should happen, this means nothing responded, but maybe if
716  // something vanishes during the search it will come up.
717  if (a && nota) return false;
718 
719  if (!a && !nota){
720  if (i == alarmSearchJunction){
721  // this is our time to decide differently, we went zero last time, go one.
722  a = 1;
723  alarmSearchJunction = lastJunction;
724  }else if (i < alarmSearchJunction){
725 
726  // take whatever we took last time, look in address
727  if (alarmSearchAddress[ibyte] & ibit){
728  a = 1;
729  }else{
