172 uint8_t Adafruit_SI1145::read8(uint8_t reg) {
173  uint16_t val;
174  Wire.beginTransmission(_addr);
175  Wire.write((uint8_t)reg);
176  Wire.endTransmission();
177 
178  Wire.requestFrom((uint8_t)_addr, (uint8_t)1);
179  return Wire.read();
180 }

182 uint16_t Adafruit_SI1145::read16(uint8_t a) {
183  uint16_t ret;
184 
185  Wire.beginTransmission(_addr); // start transmission to device
186  Wire.write(a); // sends register address to read from
187  Wire.endTransmission(); // end transmission
188 
189  Wire.requestFrom(_addr, (uint8_t)2);// send data n-bytes read
190  ret = Wire.read(); // receive DATA
191  ret |= (uint16_t)Wire.read() << 8; // receive DATA
192 
193  return ret;
194 }
