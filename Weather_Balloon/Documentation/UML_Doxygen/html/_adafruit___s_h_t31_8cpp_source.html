69 boolean Adafruit_SHT31::readTempHum(void) {
70  uint8_t readbuffer[6];
71 
72  writeCommand(SHT31_MEAS_HIGHREP);
73 
74  delay(500);
75  Wire.requestFrom(_i2caddr, (uint8_t)6);
76  if (Wire.available() != 6)
77  return false;
78  for (uint8_t i=0; i<6; i++) {
79  readbuffer[i] = Wire.read();
80  // Serial.print("0x"); Serial.println(readbuffer[i], HEX);
81  }
82  uint16_t ST, SRH;
83  ST = readbuffer[0];
84  ST <<= 8;
85  ST |= readbuffer[1];
86 
87  if (readbuffer[2] != crc8(readbuffer, 2)) return false;
88 
89  SRH = readbuffer[3];
90  SRH <<= 8;
91  SRH |= readbuffer[4];
92 
93  if (readbuffer[5] != crc8(readbuffer+3, 2)) return false;
94 
95  // Serial.print("ST = "); Serial.println(ST);
96  double stemp = ST;
97  stemp *= 175;
98  stemp /= 0xffff;
99  stemp = -45 + stemp;
100  temp = stemp;
101 
102 // Serial.print("SRH = "); Serial.println(SRH);
103  double shum = SRH;
104  shum *= 100;
105  shum /= 0xFFFF;
106 
107  humidity = shum;
108 
109  return true;
110 }

119 uint8_t Adafruit_SHT31::crc8(const uint8_t *data, int len)
120 {
121 /*
122 *
123  * CRC-8 formula from page 14 of SHT spec pdf
124  *
125  * Test data 0xBE, 0xEF should yield 0x92
126  *
127  * Initialization data 0xFF
128  * Polynomial 0x31 (x8 + x5 +x4 +1)
129  * Final XOR 0x00
130  */
131 
132  const uint8_t POLYNOMIAL(0x31);
133  uint8_t crc(0xFF);
134 
135  for ( int j = len; j; --j ) {
136  crc ^= *data++;
137 
138  for ( int i = 8; i; --i ) {
139  crc = ( crc & 0x80 )
140  ? (crc << 1) ^ POLYNOMIAL
141  : (crc << 1);
142  }
143  }
144  return crc;
145 }
